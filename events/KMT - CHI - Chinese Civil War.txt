country_event = {
	id = 95001
	title = "Should We Assist the PRC?"
    desc = "The PRC have begun taking over Mainland China, should we assist them?"
    picture = "Rebellion"

    	fire_only_once = yes

    	trigger = {
    		tag = RUS
    		government = proletarian_dictatorship
    		year = 1946
    		KMT = { has_country_flag = chinese_civil_war }
    		NOT = {
    			OR = {
    			has_country_flag = soviets_assisted_prc
    			has_country_flag = soviets_rejected_prc
    			}
    		}
    		money = 100000
    	}

    	mean_time_to_happen = {
    		months = 4
    	}

    	option = {
    		name = "Assist the peoples' republic of China!"
    		set_country_flag = soviets_assisted_prc
			RUS = {
				money = -100000
				diplomatic_influence = {
                	who = CHI
                	value = 150
                }
                relation = {
                	who = CHI
                	value = 200
                }
                create_alliance = CHI
			}
			CHI = {
				money = 100000
			}
			ai_will_do = { factor = 1 }
    	}

    	option = {
    		name = "Forget it, we need to cover our own expenses!"
    		set_country_flag = soviets_rejected_prc
				diplomatic_influence = {
                	who = CHI
                	value = -150
                }
                relation = {
                	who = CHI
                	value = -200
                }
    	}
}

country_event = {
	id = 95003
	title = "Peoples' Republic of China Sends Us a Peace Deal!"
	desc = "Peoples' Republic of China sent us a peace deal demanding that we surrender to their forces! Should we accept this?"
	picture = "Rebellion"

	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		tag = KMT
		NOT = {
			government = proletarian_dictatorship
		}
		year = 1946
		KMT = {
			has_country_flag = chinese_civil_war
		}
	}

	option = {
		name = "Accept the peace deal..."
		set_country_flag = kmt_surrenders

		KMT = {
			civilized = yes
		}

		CHI = {
			inherit = KMT
			end_war = KMT
			prestige = 25

		}

		ai_will_do = {
			factor = -1
			modifier = {
				factor = 2

					controller = CHI
				}
			}
		}
	}

	option = {
		name = "We refuse to accept the peace deal!"
		KMT = {
			war_exhaustion = 10
		}

		ai_will_do = {
			factor = 0
			modifier = {
				factor = 1
				1612 = {
					controller = KMT
				}
			}
		}
	}
}
